016package org.fusesource.jansi;
017
018import static org.fusesource.jansi.internal.CLibrary.STDERR_FILENO;
019import static org.fusesource.jansi.internal.CLibrary.STDOUT_FILENO;
020import static org.fusesource.jansi.internal.CLibrary.isatty;
021
022import java.io.FilterOutputStream;
023import java.io.IOException;
024import java.io.OutputStream;
025import java.io.PrintStream;
026
027/**
028 * Provides consistent access to an ANSI aware console PrintStream.
029 *
030 * @author <a href="http://hiramchirino.com">Hiram Chirino</a>
031 * @since 1.0
032 */
033public class AnsiConsole {
034
035    public static final PrintStream system_out = System.out;
036    public static final PrintStream out = new PrintStream( wrapOutputStream( system_out ) );
037
038    public static final PrintStream system_err = System.err;
039    public static final PrintStream err = new PrintStream( wrapErrorOutputStream( system_err ) );
040
041    private static int installed;
042
043    private AnsiConsole() {
044    }
045
046    public static OutputStream wrapOutputStream(final OutputStream stream) {
047        try {
048            return wrapOutputStream(stream, STDOUT_FILENO);
049        } catch (Throwable ignore) {
050            return wrapOutputStream(stream, 0);
051        }
052    }
053
054    public static OutputStream wrapErrorOutputStream(final OutputStream stream) {
055        try {
056            return wrapOutputStream(stream, STDERR_FILENO);
057        } catch (Throwable ignore) {
058            return wrapOutputStream(stream, 0);
059        }
060    }
061
062    public static OutputStream wrapOutputStream(final OutputStream stream, int fileno) {
063
064        // If the jansi.passthrough property is set, then don't interpret
065        // any of the ansi sequences.
066        if (Boolean.getBoolean("jansi.passthrough")) {
067            return stream;
068        }
069
070        // If the jansi.strip property is set, then we just strip the
071        // the ansi escapes.
072        if (Boolean.getBoolean("jansi.strip")) {
073            return new AnsiOutputStream(stream);
074        }
075
076        String os = System.getProperty("os.name");
077        if (os.startsWith("Windows") && !isXterm()) {
078
079            // On windows we know the console does not interpret ANSI codes..
080            try {
081                return new WindowsAnsiOutputStream(stream);
082            } catch (Throwable ignore) {
083                // this happens when JNA is not in the path.. or
084                // this happens when the stdout is being redirected to a file.
085            }
086
087            // Use the ANSIOutputStream to strip out the ANSI escape sequences.
088            return new AnsiOutputStream(stream);
089        }
090
091        // We must be on some Unix variant, including Cygwin or MSYS(2) on Windows...
092        try {
093            // If the jansi.force property is set, then we force to output
094            // the ansi escapes for piping it into ansi color aware commands (e.g. less -r)
095            boolean forceColored = Boolean.getBoolean("jansi.force");
096            // If we can detect that stdout is not a tty.. then setup
097            // to strip the ANSI sequences..
098            if (!isXterm() && !forceColored && isatty(fileno) == 0) {
099                return new AnsiOutputStream(stream);
100            }
101        } catch (Throwable ignore) {
102            // These errors happen if the JNI lib is not available for your platform.
103            // But since we are on ANSI friendly platform, assume the user is on the console.
104        }
105
106        // By default we assume your Unix tty can handle ANSI codes.
107        // Just wrap it up so that when we get closed, we reset the
108        // attributes.
109        return new FilterOutputStream(stream) {
110            @Override
111            public void close() throws IOException {
112                write(AnsiOutputStream.REST_CODE);
113                flush();
114                super.close();
115            }
116        };
117    }
118
119    private static boolean isXterm() {
120        String term = System.getenv("TERM");
121        return term != null && term.startsWith("xterm");
122    }
123
124    /**
125     * If the standard out natively supports ANSI escape codes, then this just
126     * returns System.out, otherwise it will provide an ANSI aware PrintStream
127     * which strips out the ANSI escape sequences or which implement the escape
128     * sequences.
129     *
130     * @return a PrintStream which is ANSI aware.
131     */
132    public static PrintStream out() {
133        return out;
134    }
135
136    /**
137     * If the standard out natively supports ANSI escape codes, then this just
138     * returns System.err, otherwise it will provide an ANSI aware PrintStream
139     * which strips out the ANSI escape sequences or which implement the escape
140     * sequences.
141     *
142     * @return a PrintStream which is ANSI aware.
143     */
144    public static PrintStream err() {
145        return err;
146    }
147
148    /**
149     * Install Console.out to System.out.
150     */
151    synchronized static public void systemInstall() {
152        installed++;
153        if (installed == 1) {
154            System.setOut(out);
155            System.setErr(err);
156        }
157    }
158
159    /**
160     * undo a previous {@link #systemInstall()}.  If {@link #systemInstall()} was called
161     * multiple times, it {@link #systemUninstall()} must call the same number of times before
162     * it is actually uninstalled.
163     */
164    synchronized public static void systemUninstall() {
165        installed--;
166        if (installed == 0) {
167            System.setOut(system_out);
168            System.setErr(system_err);
169        }
170    }
171
172}
